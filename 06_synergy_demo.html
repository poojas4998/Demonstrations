find_combined_markers_singletreatment <- function(seurat_object, all_combinations,group_column = "sample",control_name = "PBS") {
  # seurat_nk_cells need to be a seurat object
  DefaultAssay(seurat_object) <- "RNA"
  cluster_results <- lapply(all_combinations, function(cluster_id) {
    markers <- FindMarkers(seurat_object,
                           ident.1 = cluster_id,
                           ident.2 = control_name,
                           group.by = group_column,
                           verbose = FALSE) %>%
      rownames_to_column(var = "gene_name") %>%
      mutate(
        Comparison = paste(cluster_id, "vs." , paste0(control_name)),
        updated_condition = cluster_id
      ) 
    return(as.data.frame(markers))
  })
  combined_markers <- do.call(rbind, cluster_results)
  return(combined_markers)
}

subset_genes <- function(df, p_val_adj_threshold = 0.05, project_folder = "DEG_exports") {
  # Step 1: Subset genes
  upregulated <- df %>% filter(avg_log2FC > 0.5 & p_val_adj < p_val_adj_threshold)
  downregulated <- df %>% filter(avg_log2FC < -0.5 & p_val_adj < p_val_adj_threshold)
  significant <- df %>% filter(abs(avg_log2FC) > 0.5 & p_val_adj < p_val_adj_threshold)
  
  # Step 2: Prepare folder
  padj_name <- sprintf("%03d", as.numeric(gsub("\\.", "", as.character(p_val_adj_threshold * 100))))
  # output_folder <- file.path(project_folder, paste0("padj_", padj_name))
  output_folder <- file.path(paste0("padj_", padj_name))
  dir.create(output_folder, recursive = TRUE, showWarnings = FALSE)
  
  # Step 3: Save combined CSVs
  write.csv(upregulated, file = file.path(output_folder, "upsigDEGs.csv"), row.names = FALSE)
  write.csv(downregulated, file = file.path(output_folder, "downsigDEGs.csv"), row.names = FALSE)
  write.csv(significant, file = file.path(output_folder, "allsigDEGs.csv"), row.names = FALSE)
  
  # Step 4: Helper to split by updated_condition and export
  split_sort_save <- function(df, category_suffix) {
    df_list <- df %>%
      split(.$updated_condition) %>%
      map(~ arrange(.x, desc(avg_log2FC)))

    # Save each condition's file
    walk2(df_list, names(df_list), function(data, condition) {
      safe_name <- paste0(gsub("[^a-zA-Z0-9]", "_", condition), "_", category_suffix)
      write.csv(data, file = file.path(output_folder, paste0(safe_name, ".csv")), row.names = FALSE)
    })
    
    return(df_list)
  }
  
  # Step 5: Split, save, and return all in a structured list
  result <- list(
    combined = list(
      upregulated = upregulated,
      downregulated = downregulated,
      significant = significant
    ),
    per_condition = list(
      upsig = split_sort_save(upregulated, "upsig"),
      downsig = split_sort_save(downregulated, "downsig"),
      sig = split_sort_save(significant, "sig")
    )
  )
  
  return(result)
}

move_files_to_synergy <- function(target_folder, files) {
  if (grepl("^PBS", target_folder)) {
    cat("[SKIPPING] Folder starts with 'PBS':", target_folder, "\n")
    return()
  }
  
  target_folder_path <- file.path("synergy", target_folder)
  if (!dir.exists(target_folder_path)) {
    dir.create(target_folder_path, recursive = TRUE)
  }

  for (file in files) {
    original_file_path <- file.path(folder_path, file)
    
    if (file.exists(original_file_path)) {
      output_path <- file.path(target_folder_path, file)
      
      file.copy(original_file_path, output_path, overwrite = TRUE)
      cat("[MOVED] File:", file, "->", output_path, "\n")
    } else {
      cat("[MISSING] File not found:", file, "\n")
    }
  }
}

  if (file.exists(file_path)) {

Synergy test (Z > X + Y):
We formalize synergy for a combination Z = (12R+4R or 4R+12R) vs its components X and Y by computing per-gene mean expression and flagging synergy genes where:

mean(Z)‚ÄÖ‚Ää>‚ÄÖ‚Äämean(X)+mean(Y).mean(Z)>mean(X)+mean(Y).

This highlights supra-additive transcriptional programs attributable to the combined signal, not seen with single receptors alone.

plot_synergy_barplot <- function(df_long,
                                 treatment_name,
                                 custom_colors,
                                 sample_col = "Sample",
                                 expression_col = "Expression",
                                 cytokine_col = "Cytokine",
                                 plot_title_suffix = "Decomposition by Component") {
  
  # Convert strings to symbols for tidy evaluation
  sample_sym <- sym(sample_col)
  expression_sym <- sym(expression_col)
  cytokine_sym <- sym(cytokine_col)
  
  # Ensure proper factor levels for Sample and Cytokine
  df_long[[sample_col]] <- factor(df_long[[sample_col]], levels = unique(df_long[[sample_col]]))
  df_long[[cytokine_col]] <- factor(df_long[[cytokine_col]], levels = c("Difference", names(custom_colors)[2:3]))
  
  # Generate stacked bar plot
  p <- ggplot(df_long, aes(x = !!sample_sym, y = !!expression_sym, fill = !!cytokine_sym)) +
    geom_bar(stat = "identity", position = "stack", width = 0.4) +
    coord_flip() +
    scale_fill_manual(values = custom_colors, name = "Component") +
    labs(
      title = paste(treatment_name, plot_title_suffix),
      subtitle = "Ordered by Total Expression (Treatment + Controls)",
      x = NULL,
      y = "Expression"
    ) +
    theme_minimal(base_size = 13) +
    theme(
      panel.grid.major.y = element_blank(),
      panel.grid.minor = element_blank(),
      axis.text.y = element_text(face = "bold"),
      plot.title = element_text(face = "bold", size = 16),
      legend.position = "bottom",
      legend.title = element_text(face = "bold"),
      legend.text = element_text(size = 11),
      axis.ticks.y = element_blank(),
      plot.margin = margin(10, 10, 10, 0)
    )
  
  return(p)
}

Why this matters for o10R biology

  • o10R phenotype: IL-10 pathway rewiring can promote a fit, persistence-prone state while retaining effector potential. In our pipeline, this is reflected by exclusive/unique DE genes under o10R contexts and positive ‚ÄúDifference‚Äù bars in the combo plots.

  • STAT programs: Although IL-10 canonically engages STAT3, crosstalk can broaden to STAT1/5 in engineered contexts. The DE and synergy lists provide concrete gene-level evidence to probe these axes (e.g., STAT target signatures, survival/exhaustion modules).

  • Translational angle: Efficacy without preconditioning is consistent with durable transcriptional programs rather than transient boosts; synergy genes are natural candidates for pathway enrichment and mechanistic validation (e.g., knockdown/overexpression, phospho-flow).

How to read the outputs

  • padj_005/*_upsig.csv ‚Äî Condition-specific upregulated genes vs 10R (use these for pathway enrichment per condition).

  • synergy/<pair>/analysis_results/common_genes.csv ‚Äî shared signature across files for that pair.

  • .../exclusive_genes_*.csv ‚Äî most characteristic genes for a file/condition.

  • <pair>_difference.png ‚Äî Supra-additive genes rank-ordered; taller Difference bars = stronger synergy beyond the sum of singles.
